hd <- seq.Date(from = as.Date("2023-03-06"), 
                  to = as.Date("2023-03-10"), 
                  by = 1) %>%
  as.data.frame() %>%
  rename(date=".") %>%
  mutate(holiday="Spring Break") %>%
  bind_rows(data.frame(date=as.Date("2023-01-16"),
                       holiday="MLK"
                       ))

Wrangling

dates_w <- data.frame(date=dates) %>%
  mutate(dow=wday(date, label = T)) %>%
  filter(dow %in% c("Mon", "Wed", "Fri")) %>%
  mutate(time=case_when(
    dow %in% c("Mon","Fri") ~ "9:10 to 10:00",
    T ~ "9:10 to 11:10"
  )) %>%
  left_join(hd)
