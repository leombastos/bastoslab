hd <- seq.Date(from = as.Date("2024-03-04"), 
                  to = as.Date("2024-03-08"), 
                  by = 1) %>%
  as.data.frame() %>%
  rename(date=".") %>%
  mutate(holiday="Spring Break") %>%
  bind_rows(data.frame(date=as.Date("2024-01-15"),
                       holiday="MLK"
                       ))

Wrangling

dates_w <- data.frame(date=dates) %>%
  mutate(dow=wday(date, label = T)) %>%
  filter(dow %in% c("Tue", "Thu")) %>%
  mutate(time=case_when(
    dow == "Tue" ~ "9:30 to 11:10",
    T ~ "9:30 to 11:30"
  )) %>%
  left_join(hd)
